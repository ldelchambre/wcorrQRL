65 double randn(const double mu, const double sigma) {
66  static int isset = 0;
67  static double z0 = 0, z1 = 0; // Dummy initialization to avoid compilation warning
68  double u0,u1;
69 
70  // If z was already computed
71  if(isset) {
72  isset = 0; // Say that we have to re-compute z0, z1
73  return z1*sigma+mu;
74  }
75 
76  // Pick up u1,u2 in [DBL_EPSILON, 1]
77  do {
78  u0 = 1./RAND_MAX*rand();
79  u1 = 1./RAND_MAX*rand();
80  } while(u0 <= DBL_EPSILON);
81 
82  // Compute the Box-Muller transform of u1, u2
83  z0 = sqrt(-2.0 * log(u0)) * cos(2.*M_PI*u1);
84  z1 = sqrt(-2.0 * log(u0)) * sin(2.*M_PI*u1);
85 
86  // Say that z1 is still usable
87  isset = 1;
88 
89  return z0 * sigma + mu;
90 }

127 void build_mock_observation(double *s, double *w, const unsigned long n, const double *T, const unsigned long nvar, const unsigned long ncomp, const unsigned long shift, const double noise) {
128  const double (*Tp)[nvar] = (const double (*)[nvar]) T;
129  double a, sigma;
130  unsigned long i,j;
131 
132  // Initialize observation
133  for(i = 0; i < n; i++)
134  s[i] = 0;
135 
136  // Compute a random linear combination of the templates at the observation
137  for(i = 0; i < ncomp; i++) {
138  // Pick-up the random linear coefficient to apply to the template
139  a = (2.* rand()/RAND_MAX-1.);
140  // Add the shifted template
141  for(j = 0; j < n; j++)
142  s[j] += Tp[i][(j + shift) % nvar] * a;
143  }
144 
145  // Add noise to the observation
146  for(i = 0; i < n; i++) {
147  sigma = noise * (1. + rand()) / (1. + RAND_MAX); // sigma = ]0, noise]
148  s[i] += randn(0.,sigma);
149  w[i] = 1./sigma;
150  }
151 }
