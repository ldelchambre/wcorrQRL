40 double *wcorr_lookup(double *L, const double *T, const unsigned long nvar, const unsigned long ncomp) {
41  const unsigned long ncol = ncomp*(ncomp+3)/2; // Number of column in the lookup table
42  const double (*Tp)[nvar] = (const double (*)[nvar]) T; // Reinterpret T as being an array of observations
43  double (*Lp)[nvar] = (double (*)[nvar]) L; // Reinterpret L as being an array of observations
44  unsigned long i,j,k,l;// General usage counters
45 
46  // Fill the lookup table with the element-wise product of the templates plus
47  // hte individuals templates
48  i = 0;
49  for(j = 0; j < ncomp; j++,i++) {
50  for(k = j; k < ncomp; k++,i++)
51  for(l = 0; l < nvar; l++)
52  Lp[i][l] = Tp[j][l] * Tp[k][l]; // L(:,i) = T(:,j) .* T(:,k)
53  for(l = 0; l < nvar; l++)
54  Lp[i][l] = Tp[j][l]; // L(:,i) = T(:,j)
55  }
56 
57  return fftr(L,L,nvar,ncol); // Return the DFT of the extended array of templates
58 }

75 void _wcorr(double *y, double *A, const double *w, const double *s, const double *T, const unsigned long nvar, const unsigned long ncomp) {
76  // Matrices used in the QR orthogonalization
77  double X[ncomp*(ncomp+1)]; // X = [(W.*T)(1:ncomp,:) (W.*S)(1:ncomp)]
78 
79  // General usage variable
80  double alpha, gamma, temp;
81  unsigned long i,j,k,l,il,z;
82 
83  // Reinterpret some matrices in order to ease understanding
84  double (*Xp)[ncomp] = (double (*)[ncomp]) X;
85  const double (*Tp)[nvar] = (const double (*)[nvar]) T;
86  double (*Ap)[nvar] = (double (*)[nvar]) A;
87 
88  // Browse all shift
89  for(z = 0; z < nvar; z++) {
90  // Build X and Y matrices
91  for(i = 0; i < ncomp; i++) {
92  for(j = 0; j < ncomp; j++)
93  Xp[i][j] = w[j] * Tp[i][(j+z)%nvar];
94  Xp[ncomp][i] = w[i] * s[i];
95  }
96 
97  // Reduce all rows of X
98  il = 0; // The column index within the lookup table
99  for(i = 0; i < ncomp; i++) {
100  alpha = Ap[il++][z];
101  // Check that the ith columns has a non zero norm
102  if(0. < alpha && DBL_EPSILON * Ap[i][z] < alpha) {
103  // Annihilate the ith column of X thanks to factorized QR algorithm
104  alpha = (Xp[i][i] < 0) ? -sqrt(alpha) : sqrt(alpha);
105  temp = (Xp[i][i] += alpha);
106  for(j = i + 1; j <= ncomp; j++) {
107  gamma = (Xp[j][i] + Ap[il++][z]/alpha) / temp;
108  for(k = i; k < ncomp; k++)
109  Xp[j][k] -= gamma * Xp[i][k];
110  }
111 
112  // Update the lookup table
113  l = il;
114  for(j = i + 1; j < ncomp; j++)
115  for(k = j; k <= ncomp; k++, l++)
116  Ap[l][z] -= Xp[j][i]*Xp[k][i];
117  } else {
118  // If the ith column of X has a zero norm then X is rank-deficient
119  Xp[ncomp][i] = 0;
120  il += ncomp - i;
121  }
122  }
123 
124  // Compute the resulting weighted cross correlation function
125  temp = 0;
126  for(i = 0; i < ncomp; i++)
127  temp += Xp[ncomp][i] * Xp[ncomp][i];
128  y[z] = temp;
129  }
130 }

146 void _wcorr_zopt(double *y, double *A, const unsigned long nvar, const unsigned long ncomp, const unsigned long zpos) {
147  // Reinterpret A matrices in order to facilitate understanding
148  double (*Ap)[nvar] = (double (*)[nvar]) A;
149 
150  // General usage variable
151  double alpha, norm2, res;
152  unsigned long i,j,k,l,next,il,z,iy;
153 
154  // Compute the cross correlation function insertion index by taking into
155  // account the shift that was applied to the observation.
156  iy = nvar - zpos;
157 
158  // Browse all shift
159  for(z = 0; z < nvar; z++) {
160  res = 0;
161  for(i = il = 0; i < ncomp; i++) {
162  // Annihilate ith column
163  norm2 = Ap[il][z];
164  next = il + ncomp - i;
165  // If ith column has a non zero length then this problem is rank deficient.
166  if(0. < norm2 && DBL_EPSILON * Ap[i][z] < norm2) {
167  // Compute the resulting weighted cross correlation function
168  res += Ap[next][z]*Ap[next][z]/norm2;
169  // Update the lookup table
170  for(j = 1, l = next + 1; j < ncomp - i; j++) {
171  alpha = Ap[il+j][z] / norm2;
172  for(k = j; k <= ncomp - i; k++, l++)
173  Ap[l][z] -= alpha * Ap[il+k][z];
174  }
175  }
176  il = next + 1;
177  }
178  y[iy] = res;
179  iy = (iy + 1) % nvar;
180  }
181 }

183 double *wcorr(double *corr, const double *L, const double *T, const unsigned long nvar, const unsigned long ncomp, const double *W, const double *S, const unsigned long n, const unsigned long nobs, const int search_zopt) {
184  // Reinterpret some matrices in order to ease understanding
185  const unsigned long nfft = fftr_size(nvar,1);
186  const double (*Lp)[nfft] = (const double (*)[nfft]) L;
187  double (*corrp)[nvar] = (double (*)[nvar]) corr;
188 
189  const double *Sp; // Pointer to the spectrum of the current observation
190  const double *Wp; // Pointer to the weights associated with Sp
191  double (*FA)[nfft]; // Pointer to the inner product table in the Fourier domain
192  double (*A)[nvar];// Pointer to the inner product table in the time domain
193  double (*FY)[nfft]; // Pointer to the image matrix in the Fourier domain
194  double (*Y)[nvar]; // Pointer to the image matrix in the time domain
195  unsigned long obs; // The current observation index
196 
197  // Zero weights optimization variables
198  unsigned long zpos;
199  int zopt;
200 
201  // General usage variables
202  unsigned long i,j,k,l;
203  double temp;
204 
205  // Check that the number of variable within the observations is less or equal than the number of variables within the templates
206  if(nvar < n)
207  return NULL;
208 
209  // Allocate matrices
210  FA = (double (*)[nfft]) wcorr_lookup_alloc(nvar, ncomp);
211  FY = (double (*)[nfft]) fft_alloc(sizeof(double) * 2 * nfft);
212  A = (double (*)[nvar]) FA;
213  Y = (double (*)[nvar]) FY;
214  if(!FA || !FY) {
215  wcorr_lookup_free((double *) FA);
216  fft_free(FY);
217  return NULL;
218  }
219 
220  // For each observation
221  for(obs = 0; obs < nobs; obs++) {
222  // Initialize Sp and Wp for this observation
223  Sp = &S[obs*n];
224  Wp = &W[obs*n];
225 
226  // Check if we can shift Sp such that we have ncomp consecutive zeros within Wp
227  zopt = 0;
228  zpos = nvar;
229  if(ncomp <= nvar - n) { // By considering the padding we have to set
230  zopt = 1;
231  zpos = n; // We have ncomp consecutive zeros beyond Sp
232  } else if(search_zopt) { // Otherwise, try to find ncomp consecutive zeros within Wp (if search_zopt != 0)
233  i = 0;
234  while(i < n && !zopt) {
235  // Browse all zeros from i
236  j = i;
237  while(j < n && Wp[j] == 0 && j - i < ncomp)
238  j++;
239  // We find ncomp consecutive zeros
240  if(j - i == ncomp) {
241  zopt = 1;
242  zpos = i;
243  } else
244  i = j + 1;
245  }
246  }
247  // Here:
248  // if zopt == 1 => Wp[zpos:zpos+ncomp-1] == 0
249 
250  // Copy the square weights and weighted observation into Y by taking into
251  // account the potential shift leading to the 'zero-optimized' mode
252  if(zopt) {
253  for(i = nvar - zpos, j = 0; j < zpos; i++, j++) {
254  temp = Wp[j]*Wp[j];
255  Y[0][i] = temp;
256  Y[1][i] = temp*Sp[j];
257  }
258  for(i = 0; i < ncomp; i++)
259  Y[0][i] = Y[1][i] = 0;
260  j += ncomp;
261  } else
262  i = j = 0;
263  for(; j < n; i++, j++) {
264  temp = Wp[j]*Wp[j];
265  Y[0][i] = temp;
266  Y[1][i] = temp*Sp[j];
267  }
268 
269  // Pad Y with 0
270  for(; j < nvar; i++, j++)
271  Y[0][i] = Y[1][i] = 0;
272 
273  // Convert Y to frequency domain
274  fftr((double *) FY, (const double *) Y, nvar, 2);
275 
276  // Build the inner product table A in the frequency domain
277  i = 0;
278  for(j = 0; j < ncomp; j++) {
279  for(k = j; k < ncomp; k++) {
280  for(l = 0; l < nfft; l += 2) {
281  // (FA[i][l] + FA[i][l+1] i) = conj(FY[0][l]+FY[0][l+1] i) * (Lp[i][l] * Lp[i][l+1] i)
282  FA[i][l] = FY[0][l]*Lp[i][l] + FY[0][l+1]*Lp[i][l+1];
283  FA[i][l+1] = FY[0][l]*Lp[i][l+1] - FY[0][l+1]*Lp[i][l];
284  }
285  i++;
286  }
287  for(l = 0; l < nfft; l += 2) {
288  // (FA[i][l] + FA[i][l+1] i) = conj(FY[1][l]+FY[1][l+1] i) * (Lp[i][l] * Lp[i][l+1] i)
289  FA[i][l] = FY[1][l]*Lp[i][l] + FY[1][l+1]*Lp[i][l+1];
290  FA[i][l+1] = FY[1][l]*Lp[i][l+1] - FY[1][l+1]*Lp[i][l];
291  }
292  i++;
293  }
294 
295  // Convert the inner product table into the time domain
296  ifftr((double *) A, (const double *) FA, nvar, i);
297 
298  // Execute the factorized QR algorithm with lookup table algorithm
299  if(zopt)
300  _wcorr_zopt(corrp[obs], (double *) A, nvar, ncomp, zpos); // In the 'zero-optimized' mode
301  else
302  _wcorr(corrp[obs], (double *) A, Wp, Sp, T, nvar, ncomp); // In the classical mode
303  }
304 
305  // Free temporary arrays
306  wcorr_lookup_free((double *) FA);
307  fft_free(FY);
308 
309  return corr;
310 }
